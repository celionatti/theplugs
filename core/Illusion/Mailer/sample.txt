<?php

require_once 'mailer.php';

use Mail\Mail;

// 1. Configure the mailer (optional - has defaults)
Mail::configure([
    'default' => 'smtp',
    'drivers' => [
        'smtp' => [
            'host' => 'smtp.gmail.com',
            'port' => 587,
            'username' => 'your-email@gmail.com',
            'password' => 'your-app-password',
            'encryption' => 'tls',
            'auth' => true
        ],
        'native' => []
    ],
    'templates' => [
        'path' => __DIR__ . '/views/emails'
    ]
]);

// 2. Simple HTML email
try {
    $sent = Mail::to('recipient@example.com', 'John Doe')
        ->from('sender@myapp.com', 'My Application')
        ->subject('Simple HTML Email')
        ->html('<h1>Hello!</h1><p>This is a <strong>simple</strong> HTML email.</p>')
        ->send();
    
    echo $sent ? "Email sent successfully!\n" : "Failed to send email.\n";
} catch (Exception $e) {
    echo "Error: " . $e->getMessage() . "\n";
}

// 3. Template-based email with data
$user = [
    'name' => 'Jane Smith',
    'email' => 'jane@example.com'
];

$loginUrl = 'https://myapp.com/login';
$appName = 'My Awesome App';

try {
    Mail::to($user['email'], $user['name'])
        ->from('welcome@myapp.com', 'Welcome Team')
        ->subject('Welcome to ' . $appName)
        ->view('welcome', [
            'user' => $user,
            'loginUrl' => $loginUrl,
            'appName' => $appName
        ])
        ->send();
} catch (Exception $e) {
    echo "Welcome email error: " . $e->getMessage() . "\n";
}

// 4. Multi-recipient email with both HTML and text versions
try {
    Mail::to('user1@example.com', 'User One')
        ->to('user2@example.com', 'User Two')
        ->cc('manager@example.com', 'Manager')
        ->bcc('archive@example.com')
        ->from('newsletter@myapp.com', 'Newsletter')
        ->replyTo('support@myapp.com', 'Support Team')
        ->subject('Monthly Newsletter')
        ->view('newsletter', ['month' => 'August', 'articles' => $articles])
        ->textView('newsletter-text', ['month' => 'August', 'articles' => $articles])
        ->send();
} catch (Exception $e) {
    echo "Newsletter error: " . $e->getMessage() . "\n";
}

// 5. Email with attachments
try {
    Mail::to('client@example.com', 'Client Name')
        ->from('invoices@myapp.com', 'Billing Department')
        ->subject('Invoice #12345')
        ->view('invoice', ['invoice' => $invoiceData])
        ->attach('/path/to/invoice.pdf', 'Invoice-12345.pdf', 'application/pdf')
        ->attach('/path/to/terms.pdf', 'Terms-and-Conditions.pdf')
        ->send();
} catch (Exception $e) {
    echo "Invoice email error: " . $e->getMessage() . "\n";
}

// 6. Account verification email
$verificationToken = bin2hex(random_bytes(32));
$verifyUrl = "https://myapp.com/verify?token={$verificationToken}";

try {
    Mail::to($user['email'], $user['name'])
        ->from('noreply@myapp.com', 'Account Verification')
        ->subject('Please verify your email address')
        ->view('verify-account', [
            'user' => $user,
            'verifyUrl' => $verifyUrl
        ])
        ->textView('verify-account-text', [
            'user' => $user,
            'verifyUrl' => $verifyUrl
        ])
        ->send();
} catch (Exception $e) {
    echo "Verification email error: " . $e->getMessage() . "\n";
}

// 7. Using different drivers
try {
    // Force use of native mail() function
    Mail::to('test@example.com')
        ->from('test@myapp.com')
        ->subject('Test with Native Driver')
        ->text('This email was sent using PHP\'s native mail() function.')
        ->driver('native')
        ->send();
        
    // Force use of SMTP
    Mail::to('test@example.com')
        ->from('test@myapp.com')
        ->subject('Test with SMTP Driver')
        ->html('<p>This email was sent using SMTP.</p>')
        ->driver('smtp')
        ->send();
} catch (Exception $e) {
    echo "Driver test error: " . $e->getMessage() . "\n";
}

// 8. Raw usage without static methods (for dependency injection)
use Mail\MailManager;
use Mail\Templates\SimpleTemplateRenderer;

$config = [
    'default' => 'smtp',
    'drivers' => [
        'smtp' => [
            'host' => 'localhost',
            'port' => 587,
            'username' => '',
            'password' => '',
            'encryption' => 'tls'
        ]
    ],
    'templates' => ['path' => 'views/emails']
];

$mailManager = new MailManager($config);

try {
    $mailManager->to('recipient@example.com')
        ->from('sender@example.com')
        ->subject('Dependency Injection Example')
        ->html('<p>This shows how to use the mailer with DI.</p>')
        ->send();
} catch (Exception $e) {
    echo "DI example error: " . $e->getMessage() . "\n";
}
